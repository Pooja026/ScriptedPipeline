pipeline {

                bat "rmdir  /s /q  ScriptedPipeline"

                //bat "mvn clean -f ScriptedPipeline"

        }
        stage('install') {
            steps {
                bat "mvn install -f ScriptedPipeline"

        }
        stage('Test') {
            steps {
                bat "mvn test -f ScriptedPipeline"
            }
        }
        stage('package') {
            steps {
                bat "mvn package -f ScriptedPipeline"
            }
        }
    }
}
